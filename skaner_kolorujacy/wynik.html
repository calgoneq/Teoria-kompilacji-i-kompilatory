# Demonstruje wszystkie kategorie tokenów

/* Funkcja obliczająca silnię
   z wykorzystaniem pętli while */
fun silnia(n) {
    let wynik = 1;
    let i = 1;
    while (i <= n) {
        wynik = wynik * i;
        i = i + 1;
    }
    return wynik;
}

# Funkcja ze strzałką i warunkami
fun sprawdz(x) -> {
    if (x > 0 and x != 100) {
        print("Liczba dodatnia");
        return true;
    } else {
        print("Liczba niedodatnia lub równa 100");
        return false;
    }
}
